###########
pKa options
###########

.. indigo_option::
    :name: pKa-model
    :type: string
    :short: Sets type of pKa model (simple or advanced)
    :default: simple

.. indigo_option::
    :name: pKa-model-level
    :type: int
    :short: Sets maximal level of advanced pKa model is used for pKa estimation.
    :default: 0

.. indigo_option::
    :name: pKa-model-min-level
    :type: int
    :short: Sets minimal level of advanced pKa model is used for pKa estimation.
    :default: 0
